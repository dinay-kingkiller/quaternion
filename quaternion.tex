\title{Quaternion Rotation}

\begin{abstract}
  Quaternions provide a unified framework that can be interpreted as elements of a division ring, hyper-complex numbers, or a four-dimensional vector space.
  Despite extensive literature, definitions and conventions for quaternion operations vary, which can complicate the translation of results into computational applications.
  This paper presents a formal treatment of quaternion algebra, defining addition, multiplication, and conjugation consistently, and demonstrates their use in representing rotations.
  We emphasize a formulation suitable for direct implementation in code, laying the foundation for extending these methods to rigid-body transformations in $\mathrm{SE}(3)$.
  Proofs of basic quaternion properties are summarized, while the focus remains on applications and computational implementation.
\end{abstract}

\section{The Quaternion Ring}
Quaternions arise as part of a remarkable sequence of algebraic extensions of the real numbers.
Starting from the reals $\mathbb{R}$ and complex numbers $\mathbb{C}$, one can ask how far the familiar properties of arithmetic can be preserved in higher dimensions.
The quaternions $\mathbb{H}$ are the smallest extension in which division remains possible but commutativity of multiplication must be abandoned.
Beyond this point, even more fundamental properties, such as associativity, begin to fail, as seen in the octonions.
\begin{definition}[Quaternion]
  Formally, a quaternion is an ordered 4-tuple.
  \begin{equation}
    \mathrm{q} = (x, y, z, w) \in \mathbb{H} \subset \mathbb{R}^4.
  \end{equation}
  The set of quaternions $\mathbb{H}$, equipped with the operations $\oplus$ and $\otimes$, forms a division ring.
\end{definition}

\begin{definition}[Quaternion Addition]
  For two quaternions, addition is defined componentwise
  \begin{equation}
    \begin{pmatrix}
      x_1 \\
      y_1 \\
      z_1 \\
      w_1
    \end{pmatrix} \oplus
    \begin{pmatrix}
      x_2 \\
      y_2 \\
      z_2 \\
      w_2
    \end{pmatrix} =
    \begin{pmatrix}
      x_1 + x_2 \\
      y_1 + y_2 \\
      z_1 + z_2 \\
      w_1 + w_2
    \end{pmatrix}
  \end{equation}
\end{definition}

\begin{definition}[Quaternion Multiplication]
  Multiplication is defined as
  \begin{equation} \label{eqn:multiplication}
    \begin{pmatrix}
      x_1 \\
      y_1 \\
      z_1 \\
      w_1
    \end{pmatrix} \otimes 
    \begin{pmatrix}
      x_2 \\
      y_2 \\
      z_2 \\
      w_2
    \end{pmatrix} =
    \begin{pmatrix}
      w_1x_2 + x_1w_2 + y_1z_2 - z_1y_2 \\
      w_1y_2 - x_1z_2 + y_1w_2 + z_1x_2 \\
      w_1z_2 + x_1y_2 - y_1x_2 + z_1w_2 \\
      w_1w_2 - x_1x_2 - y_1y_2 - z_1z_2
    \end{pmatrix}
  \end{equation}
  This definition aligns with the conventions used in the ROS package \texttt{tf2}.
  While other sources may define quaternion multiplication differently, they share the same underlying algebraic structure.
  An equivalent formulation appears in the next section.
\end{definition}

\begin{theorem}
  Quaternions form a commutative group under addition, which gives
  \begin{itemize}
    \item Associativity: $\mathrm{q}_1 \oplus (\mathrm{q}_2 \oplus \mathrm{q}_3) = (\mathrm{q}_1 \oplus \mathrm{q}_2) \oplus \mathrm{q}_3$
    \item Commutativity: $\mathrm{q}_1 \oplus \mathrm{q}_2 = \mathrm{q}_2 \oplus \mathrm{q}_1$
    \item Identity: $\mathrm{q} \oplus \mathrm{0} = \mathrm{q} = \mathrm{0} \oplus \mathrm{q}$ where $\mathrm{e}_0$ is the 0-quaternion $(0, 0, 0, 0)$
    \item Inverse: $\mathrm{q} \oplus (-\mathrm{q}) = \mathrm{q} \ominus \mathrm{q} = \mathrm{e}_0$ where the inverse is defined $-\mathrm{q}=(-x, -y, -z, -w)$
  \end{itemize}
\end{theorem}

\begin{theorem}
  The (non-zero) quaternions $\mathbb{H}\setminus{\mathrm{e}_0}$ form a \emph{non-commutative} group under multiplication, which gives it the following properties
  \begin{itemize}
    \item Associativity: $\mathrm{q}_1 \otimes (\mathrm{q}_2 \otimes \mathrm{q}_3) = (\mathrm{q}_1 \otimes \mathrm{q}_2) \otimes \mathrm{q}_3$
    \item Identity: $\mathrm{q} \otimes \mathrm{1} = \mathrm{q} = \mathrm{1} \otimes \mathrm{q}$ where $\mathrm{e}_1$ is the identity quaternion $(0, 0, 0, 1)$
  \item Inverse: $\mathrm{q} \otimes \mathrm{q}^{-1} = \mathrm{e}_1 = \mathrm{q}^{-1} \otimes \mathrm{q}$ unless $\mathrm{q}$ is the zero quaternion.
  \end{itemize}
\end{theorem}

\begin{theorem}
  With the combination of operations, quaternions form a division ring, which adds distributive properties
  \begin{itemize}
    \item Left distributivity: $\mathrm{q}_1 \otimes \left(\mathrm{q}_2 \oplus \mathrm{q}_3\right) = \left(\mathrm{q}_1 \otimes \mathrm{q}_2\right) \oplus \left(\mathrm{q}_1 \otimes \mathrm{q}_3\right)$
    \item Right distributivity: $\left(\mathrm{q}_1 \oplus \mathrm{q}_2\right) \otimes \mathrm{q}_3 = \left(\mathrm{q}_1 \otimes \mathrm{q}_3\right) \oplus \left(\mathrm{q}_2 \otimes \mathrm{q}_3\right)$
  \end{itemize}
\end{theorem}

\section{The Quaternion Space}
While the component form is convenient for computation, it obscures the algebraic structure.
By introducing a basis, quaternion multiplication can be expressed in a more mnemonic and structured form.
Scalar multiplication equips quaternions with the structure of a vector space over the real numbers.
\begin{definition}[Quaternion Scaling]
  Scaling a quaternion by a real number is defined componentwise.
  \begin{equation}
    s
    \begin{pmatrix}
      x \\
      y \\
      z \\
      w
    \end{pmatrix} =
    \begin{pmatrix}
      sx \\
      sy \\
      sz \\
      sw
    \end{pmatrix}
  \end{equation}
\end{definition}

\begin{theorem}
  Quaternions form a vector space over the real numbers, which means they have the following properties
  \begin{itemize}
  \item $(\mathbb{H}, \oplus)$ forms a commutative group.
    The details of which are defined in the previous section.
  \item Compatibility: $s_1(s_2\mathrm{q}) = (s_1s_2)\mathrm{q}$
  \item Scalar Identity: $1\mathrm{q} = \mathrm{q}$ where $1\in\mathbb{R}$
  \item Distributivity over quaternion addition: $s(\mathrm{q}_1 \oplus \mathrm{q}_2) = s\mathrm{q}_1 \oplus s\mathrm{q}_2$
  \item Distributivity over scalar multiplication: $(s_1+ s_2) \mathrm{q} = s_1\mathrm{q} \oplus s_2\mathrm{q}$
  \end{itemize}
\end{theorem}

\begin{theorem}[Canonical Embedding]
  The real numbers are embedded in the quaternions, with
  \begin{equation}
    \mathbb{R} \cong \left{ s\mathrm{e}_1 \mid s\in \mathbb{R} \right} \subset \mathbb{H}.
  \end{equation}
  and this subalgebra is isomorphic to $\mathbb{R}$.
\end{theorem}

\begin{remark}
  With this theorem, and the above properties, we begin shedding the notation we've built up for simplicity. The algebra $(\mathbb{H}, \oplus, \otimes, \mathrm{e}_0, \mathrm{e}_1)$ will simply be known as $(\mathbb{H}, +, \cdot, 0, 1)$ with multiplication typically written $\mathbf{p}\mathbf{q}$ to avoid notation collisions with the dot product.
\end{remark}

\begin{definition}
  In the quaternion vector space, define four orthogonal unit basis elements
  \begin{equation}
    \mathrm{i} =
    \begin{pmatrix}
      1 \\
      0 \\
      0 \\
      0
    \end{pmatrix},
    \mathrm{j} =
    \begin{pmatrix}
      0 \\
      1 \\
      0 \\
      0
    \end{pmatrix}, 
    \mathrm{k} =
    \begin{pmatrix}
      0 \\
      0 \\
      1 \\
      0
    \end{pmatrix}, 
    \mathrm{e}_1 =
    \begin{pmatrix}
      0 \\
      0 \\
      0 \\
      1
    \end{pmatrix}.
  \end{equation}
  Every quaternion can be described in this basis
  \begin{equation}
    \mathrm{q} = x \mathrm{i} + y \mathrm{j} + z \mathrm{k} + w \mathrm{e}_1
  \end{equation}
  This representation is equivalent to the component form $(x, y, z, w)$.
\end{definition}

\begin{theorem}
  The definition of quaternion multiplication in equation \ref{eqn:multiplication} is not particularly mnemonic.
  Historically quaternions were used to extend complex numbers and as such, using the basis definition above, the complex $i^2=-1$ can be extended for quaternion hypercomplex numbers
  \begin{equation}
    \mathrm{ii} = \mathrm{jj} = \mathrm{kk} = \mathrm{ijk} = -\mathrm{e}_1
  \end{equation}
  Some papers use different definitions of unit bases, which would lead to a different equation \ref{eqn:multiplication}.
  The algebraic structure remains equivalent across differing conventions, but comparing sources can introduce sign errors or apparent changes in commutativity.
\end{theorem}

\section{Pure Quaternions}
In the previous section we said quaternions form a vector space, so technically every quaternion is a 4-vector.
In practice, quaternions are particularly useful because of how they interact with 3-vectors.
\begin{definition}[Vector Quaternion]
  Quaternions can be interpreted as vectors with an attached scalar component.
  Using a slight abuse of notation, every quaternion can be decomposed into a \emph{vector} (or \emph{pure quaternion}) $\mathbf{v} = \left(x, y, z, 0\right)$ and a scalar quaternion $s = \left(0,0,0,s\right)$: either $\mathbf{v}+s$ or, like below, $(\mathbf{v}, s)$.
  Addition is defined as
  \begin{equation}
    \begin{pmatrix}
      \mathbf{v}_1 \\
      s_1
    \end{pmatrix}
    \oplus
    \begin{pmatrix}
      \mathbf{v}_2 \\
      s_2
    \end{pmatrix}
    =
    \begin{pmatrix}
      \mathbf{v}_1 + \mathbf{v}_2\\
      s_1 + s_2
    \end{pmatrix} \nonumber
  \end{equation}
  and multiplication can be defined from the vector dot and cross product
  \begin{equation}
    \begin{pmatrix}
      \mathbf{v}_1 \\
      s_1
    \end{pmatrix}
    \otimes
    \begin{pmatrix}
      \mathbf{v}_2 \\
      s_2
    \end{pmatrix}
    =
    \begin{pmatrix}
      s_1\mathbf{v}_2 + s_2\mathbf{v}_1 + \mathbf{v}_1 \times \mathbf{v}_2 \\
      s_1s_2 - \mathbf{v}_1 \cdot \mathbf{v}_2
    \end{pmatrix} \nonumber
  \end{equation}
  For pure quaternions, that multiplication becomes
  \begin{equation}
    \begin{pmatrix}
      \mathbf{v}_1 \\
      0
    \end{pmatrix}
    \otimes
    \begin{pmatrix}
      \mathbf{v}_2 \\
      0
    \end{pmatrix} =
    \begin{pmatrix}
      \mathbf{v}_1 \times \mathbf{v}_2\\
      -\mathbf{v}_1 \cdot \mathbf{v}_2
    \end{pmatrix} \nonumber
  \end{equation}
  This shows that quaternion multiplication decomposes into a combination of the familiar dot and cross products on $\mathbb{R}^3$.
\end{definition}

\begin{theorem}
  The scalar quaternions $\Re(\mathbb{H})$ are isomorphic to $\mathbb{R}$. They are defined with the relation with the conjugate
  \begin{equation}
    \mathbb{R} \cong \Re(\mathbb{H}) := \left\{\mathrm{q}=\mathrm{q}^*\right\}
  \end{equation}
\end{theorem}

\begin{theorem}
  The pure quaternions $\Im(\mathbb{H})$ are isomorphic to $\mathbb{R}^3$. They are defined with the relation with the conjugate
  \begin{equation}
    \mathbb{R}^3 \cong \Im(\mathbb{H}) := \left\{\mathrm{q}+\mathrm{q}^*=0\right\}
  \end{equation}
\end{theorem}

\begin{theorem}
  Quaternion multiplication is not commutative.
  The commutator is defined by
  \begin{equation}
    \left[\mathrm{q}_1, \mathrm{q}_2\right] := \mathrm{q}_1 \otimes \mathrm{q}_2 - \mathrm{q}_2 \otimes \mathrm{q}_1.
  \end{equation}
  It yields pure quaternions and is given by
  \begin{equation}
    \left[
    \begin{pmatrix}
      \mathbf{v}_1 \\
      s_1
    \end{pmatrix} ,
    \begin{pmatrix}
      \mathbf{v}_2 \\
      s_2
    \end{pmatrix}
    \right] =
    \begin{pmatrix}
      \mathbf{v}_1 \\
      s_1
    \end{pmatrix} \otimes
    \begin{pmatrix}
      \mathbf{v}_2 \\
      s_2
    \end{pmatrix} -
    \begin{pmatrix}
      \mathbf{v}_2 \\
      s_2
    \end{pmatrix} \otimes
    \begin{pmatrix}
      \mathbf{v}_1 \\
      s_1
    \end{pmatrix} =
    \begin{pmatrix}
      2\mathbf{v}_1 \times \mathbf{v}_2 \\
      0
    \end{pmatrix}. \label{eqn:commutator}
  \end{equation}
\end{theorem}
\begin{proof}
  For $\mathrm{q}_1=(\mathbf{v}_1, s_1)$ and $\mathrm{q}_2=(\mathbf{v}_2, s_2)$, the commutator expands to
  \begin{equation}
    \begin{pmatrix}
      \mathbf{v}_1 \\
      s_1
    \end{pmatrix} \otimes
    \begin{pmatrix}
      \mathbf{v}_2 \\
      s_2
    \end{pmatrix} -
    \begin{pmatrix}
      \mathbf{v}_2 \\
      s_2
    \end{pmatrix} \otimes
    \begin{pmatrix}
      \mathbf{v}_1 \\
      s_1
    \end{pmatrix} =
    \begin{pmatrix}
      s_1\mathbf{v}_2 + s_2\mathbf{v}_1 + \mathbf{v}_1 \times \mathbf{v}_2 \\
      s_1s_2 - \mathbf{v}_1 \cdot \mathbf{v}_2
    \end{pmatrix} -
    \begin{pmatrix}
      s_2\mathbf{v}_1 + s_1\mathbf{v}_2 + \mathbf{v}_2 \times \mathbf{v}_1 \\
      s_2s_1 - \mathbf{v}_2 \cdot \mathbf{v}_1
    \end{pmatrix} \nonumber
  \end{equation}
  Since only the cross product is \emph{not} commutative, the commutator reduces down to
  \begin{equation}
    \begin{pmatrix}
      \mathbf{v}_1 \\
      s_1
    \end{pmatrix} \otimes
    \begin{pmatrix}
      \mathbf{v}_2 \\
      s_2
    \end{pmatrix} -
    \begin{pmatrix}
      \mathbf{v}_2 \\
      s_2
    \end{pmatrix} \otimes
    \begin{pmatrix}
      \mathbf{v}_1 \\
      s_1
    \end{pmatrix} =
    \begin{pmatrix}
      \mathbf{v}_1\times\mathbf{v}_2-\mathbf{v}_2\times\mathbf{v}_1 \\
      0
    \end{pmatrix} \nonumber
  \end{equation}
  But, because the cross product is \emph{anti-}commutative
  \begin{equation}
    \begin{pmatrix}
      \mathbf{v}_1 \\
      s_1
    \end{pmatrix} \otimes
    \begin{pmatrix}
      \mathbf{v}_2 \\
      s_2
    \end{pmatrix} -
    \begin{pmatrix}
      \mathbf{v}_2 \\
      s_2
    \end{pmatrix} \otimes
    \begin{pmatrix}
      \mathbf{v}_1 \\
      s_1
    \end{pmatrix} =
    \begin{pmatrix}
      2\mathbf{v}_1\times\mathbf{v}_2\\
      0
    \end{pmatrix} \nonumber
  \end{equation}
\end{proof}

\section{Unit Quaternion Subgroup}
The algebra of quaternions admits a natural notion of conjugation, which in turn defines a quadratic quantity known as the quadrance.
Together, these structures endow $\mathbb{H}$ with the structure of a \emph{composition algebra}, in which the quadrance is multiplicative under quaternion multiplication.
This multiplicative property allows us to identify a distinguished subset of quaternions whose quadrance is unity.
As we will see, this subset forms a group under quaternion multiplication and will play a central role in describing rotations in three-dimensional space.

\begin{definition}
  For a quaternion $\mathrm{q}=(\mathbf{v}, s)$, its \emph{conjugate} is given by
  \begin{equation}
    \mathrm{q}^* := (-\mathbf{v}, s)
  \end{equation}
\end{definition}

\begin{definition}
  The \emph{quadrance} of a quaternion is the map
  \begin{equation}
    Q: \mathbb{H} \to \Re(\mathbb{H}) \cong \mathbb{R}
  \end{equation}
  defined by
  \begin{equation}
    Q(\mathrm{q}) := \mathrm{q} \otimes \mathrm{q}^*.
  \end{equation}
  Or, in coordinate form
  \begin{equation}
    Q(x\mathrm{i}+y\mathrm{j}+z\mathrm{k}+w) = x^2 + y^2 + z^2 + w^2.
  \end{equation}
\end{definition}

\begin{theorem}
  With these new definitions, the quaternion algebra $\mathbb{H}$ forms a composition algebra, which has the following properties:
  \begin{itemize}
    \item Conjugation Involution: $\left(\mathrm{q}^*\right)^* = \mathrm{q}$,
    \item Conjugation Reverses Multiplication: $(\mathrm{p}\otimes\mathrm{q})^*=\mathrm{q}^*\otimes\mathrm{p}^*$,
    \item Quadrance Nondegeneracy: $Q(\mathrm{q})=0 \Leftrightarrow \mathrm{q} = \mathrm{0}$,
    \item Quadrance Compatibility with Multiplication: $Q(\mathrm{p})Q(\mathrm{q})=Q(\mathrm{p}\otimes \mathrm{q})$.
  \end{itemize}
\end{theorem}

\begin{definition}
  The \emph{unit quaternions}, or \emph{versors}, form the subgroup $\mathrm{Spin}(3) \subset (\mathbb{H}\setminus\{\mathrm{0}\}, \otimes)$, is the set of quaternions with quadrance equal to $1$.
  \begin{equation}
    \mathrm{Spin}(3) := \left\{\mathrm{q} \in \mathbb{H} : Q(\mathrm{q}) = 1\right\}
  \end{equation}
\end{definition}

\begin{theorem}
  This group has the following group properties
  \begin{itemize}
    \item Closure: If $Q(\mathrm{p})=1$ and $Q(\mathrm{q})=1$, then $Q(\mathrm{p}\otimes\mathrm{q})=1$.
    \item Identity: The identity quaternion is a versor $\mathrm{1}\in\mathrm{Spin}(3)$.
    \item Inverse: The inverse of a versor is the conjugate which is also a versor $\mathrm{q} \otimes \mathrm{q}^* = \mathrm{1}$.
  \end{itemize}
\end{theorem}

\section{Quaternion Rotation Group}

\begin{definition}
  Unit quaternions act on vectors by conjugation, producing rotations in three-dimensional space.
  These transformations form the rotation group.
  \begin{equation}
    \mathrm{Rot}(3) := \{\mathrm{R}_{\mathrm{q}}:\Im(\mathbb{H})\to\Im(\mathbb{H}) : \mathrm{q}\in\mathrm{Spin}(3), \mathbf{v} \mapsto \mathrm{q}\otimes (\mathbf{v}, 0) \otimes\mathrm{q}^*\}
  \end{equation}
\end{definition}

\begin{theorem}
  These transformations form a group
  \begin{itemize}
    \item Closure: Composition is quaternion multiplication $R_\mathrm{p} \circ R_\mathrm{q} = R_{\mathrm{p}\otimes\mathrm{q}}$
    \item Identity: The identity quaternion forms the identity rotation $\mathbf{v} = R_\mathrm{1}(\mathbf{v})$
    \item Inverse: The conjugate forms the inverse rotation $R_\mathrm{q} \circ R_{\mathrm{q}^*} = R_\mathrm{1}$
  \end{itemize}
\end{theorem}

\begin{theorem}
  Quaternion rotation preserves the structure of $\mathbb{R}^3$
  \begin{itemize}
    \item Preserves Quadrance: $Q(R_\mathrm{q}(\mathbf{v}))=Q(\mathbf{v})$
    \item Preserves the Dot Product: $\mathbf{u} \cdot \mathbf{v}=R_\mathrm{q}(\mathbf{u}) \cdot R_\mathrm{q}(\mathbf{v})$
    \item Preserves the Cross Product: $R_\mathrm{q}\left(\mathbf{u}\times\mathbf{v}\right)=R_\mathrm{q}(\mathbf{u})\times R_\mathrm{q}(\mathbf{v})$.
  \end{itemize}
\end{theorem}

\begin{theorem}
  Any vector can be rotated to any other vector of the same quadrance,
  \begin{equation}
    Q(\mathbf{u})=Q(\mathbf{v}) \Rightarrow \exists \mathrm{q}\in \mathrm{Spin}(3)\text{ such that }\mathbf{u}=R_\mathrm{q}(\mathbf{v}).
  \end{equation}
\end{theorem}

\begin{theorem}
  However, these rotations are not unique,
  \begin{equation}
    R_\mathrm{q} = R_{-\mathrm{q}}.
  \end{equation}
  Thus unit quaternions provide a double cover of the rotation group.
\end{theorem}

\begin{theorem}[Quaternion to Matrix]
  Every rotation $\mathrm{A}\to\mathrm{B}$ can be described as both a quaternion and a rotation matrix
  \begin{equation}
    R\mathbf{v} = R_\mathrm{q} \left(\mathbf{v}\right)
  \end{equation}
  that rotation matrix also collects the dot products of the rotation.
  
  \begin{tabular}{c|ccc}
    $R$ & $\hat{a}_x$ & $\hat{a}_y$ & $\hat{a}_z$ \\
    \hline
    $\hat{b}_x$ & $1-2(y^2+z^2)$ & $2(xy-zw)$ & $2(xz+yw)$ \\
    $\hat{b}_y$ & $2(xy+zw)$ & $1-2(x^2+z^2)$ & $2(yz-xw)$ \\
    $\hat{b}_z$ & $2(xz-wy)$ & $2(xw+yz)$ & $1-2(x^2+y^2)$
  \end{tabular}\centering
\end{theorem}
\begin{proof}
  $R$ can be formed from algebra.
  \begin{equation}
    \begin{pmatrix}
      R & \mathbf{0}^\top \\
      \mathbf{0} & 0
    \end{pmatrix}
    \begin{pmatrix}
      v_x \\
      v_y \\
      v_z \\
      0
    \end{pmatrix}
    =
    \begin{pmatrix}
      x \\
      y \\
      z \\
      w
    \end{pmatrix} \otimes
    \begin{pmatrix}
      v_x \\
      v_y \\
      v_z \\
      0
    \end{pmatrix}
    \otimes
    \begin{pmatrix}
      -x \\
      -y \\
      -z \\
      w
    \end{pmatrix} \nonumber
  \end{equation}
  Expanding and collecting like terms
  \begin{equation}
    R =
    \begin{pmatrix}
      x^2-y^2-z^2+w^2 & 2xy-2zw & 2xz+2yw \\
      2xy+2zw & -x^2+y^2-z^2+w^2 & -2xw+2yz \\
      2xz-2wy & 2xw+2yz & -x^2-y^2+z^2+w^2
    \end{pmatrix} \nonumber
  \end{equation}
  Since this is a unit quaternion, its quadrance is equal to one: $1=x^2+y^2+z^2+w^2$.
  \begin{equation}
    R =
    \begin{pmatrix}
      1-2(y^2+z^2) & 2(xy-zw) & 2(xz+yw) \\
      2(xy+zw) & 1-2(x^2+z^2) & 2(yz-xw) \\
      2(xz-wy) & 2(xw+yz) & 1-2(x^2+y^2)
    \end{pmatrix} \nonumber
  \end{equation}
\end{proof}

\begin{theorem}
  Let $\mathrm{q}=(\mathbf{q}, q_0)$ be a unit quaternion. The rotated vector $\mathbf{v}'$ can be computed as
  \begin{equation}
    \mathbf{v}' = \mathbf{v} + 2q_0\mathbf{q}\times\mathbf{v} + 2\mathbf{q}\times\left(\mathbf{q}\times\mathbf{v}\right).
  \end{equation}
\end{theorem}
\begin{proof}
  Expanding the rotation equation into its scalar and vector components
  \begin{equation}
    R_\mathrm{q}(\mathbf{v}) = \mathrm{q} \otimes (\mathbf{v}, 0) \otimes \mathrm{q}^* =
    \begin{pmatrix}
      \mathbf{q} \\
      q_0
    \end{pmatrix} \otimes
    \begin{pmatrix}
      \mathbf{v} \\
      0
    \end{pmatrix} \otimes
    \begin{pmatrix}
      -\mathbf{q} \\
      q_0
    \end{pmatrix} \nonumber
  \end{equation}
  Expanding the multiplication
  \begin{align}
    R_\mathrm{q} (\mathbf{v}) &=
    \begin{pmatrix}
      \mathbf{q} \\
      q_0
    \end{pmatrix} \otimes
    \begin{pmatrix}
      -\mathbf{v}\times\mathbf{q} + q_0\mathbf{v} \\
      \mathbf{v}\cdot\mathbf{q}
    \end{pmatrix} \nonumber \\
    &=
    \begin{pmatrix}
      \mathbf{q}\times\left(-\mathbf{v}\times\mathbf{q}\right) + q_0\mathbf{q}\times\mathbf{v} + (\mathbf{v}\cdot\mathbf{q})\mathbf{q} - q_0\mathbf{v}\times\mathbf{q} + q_0^2\mathbf{v} \\
      \mathbf{q}\cdot(\mathbf{v}\times\mathbf{q}) - q_0\mathbf{q}\cdot\mathbf{v} + q_0\mathbf{v}\cdot\mathbf{q}
    \end{pmatrix} \nonumber
  \end{align}
  Since the dot product is symmetric, and the cross product is antisymmetric, we can combine some like terms
  \begin{equation}
     R_\mathrm{q} (\mathbf{v}) =
    \begin{pmatrix}
      q_0^2\mathbf{v} + (\mathbf{v}\cdot\mathbf{q})\mathbf{q} + 2q_0\mathbf{q}\times\mathbf{v} + \mathbf{q}\times\left(\mathbf{q}\times\mathbf{v}\right) \\
      \mathbf{q} \cdot (\mathbf{v}\times\mathbf{q})
    \end{pmatrix} \nonumber
  \end{equation}
  Since $\mathrm{q}$ is a unit quaternion $q_0^2+\mathbf{q}\cdot\mathbf{q}=1$
  \begin{equation}
     R_\mathrm{q} (\mathbf{v}) =
    \begin{pmatrix}
      \mathbf{v}-(\mathbf{q}\cdot\mathbf{q})\mathbf{v} + (\mathbf{v}\cdot\mathbf{q})\mathbf{q} + 2q_0\mathbf{q}\times\mathbf{v} + \mathbf{q}\times\left(\mathbf{q}\times\mathbf{v}\right)\\
      \mathbf{q}\cdot(\mathbf{v}\times\mathbf{q})
    \end{pmatrix} \nonumber
  \end{equation}
  Using the identity $\mathbf{a}\cdot(\mathbf{a}\times\mathbf{b})=0$ and the triple cross product formula $\mathbf{a}\times(\mathbf{b}\times\mathbf{c})=(\mathbf{c}\cdot\mathbf{a})\mathbf{b}-(\mathbf{a}\cdot\mathbf{b})\mathbf{c}$
  \begin{equation}
    R_\mathrm{q}(\mathbf{v}) = \begin{pmatrix}
      \mathbf{v}+2q_0\mathbf{q}\times\mathbf{v}+2\mathbf{q}\times\left(\mathbf{q}\times\mathbf{v}\right) \\
      0
    \end{pmatrix} \nonumber
  \end{equation}
  The vector part of this new quaternion is the rotated vector $\mathbf{v}'$
  \begin{equation}
    \mathbf{v}' = \mathbf{v}+2q_0\mathbf{q}\times\mathbf{v}+2\mathbf{q}\times\left(\mathbf{q}\times\mathbf{v}\right). \nonumber
  \end{equation}
\end{proof}

\section{Quaternion Rotation Algebra}

\begin{theorem}
  The Lie algebra that corresponds the Spin group is the set of pure quaternions
  \begin{equation}
    \mathfrak{spin}(3) := \{\mathrm{\dot{q}}\in\mathbb{H}: \mathrm{\dot{q}}+\mathrm{\dot{q}}^* = 0\} = \mathrm{Im}(\mathbb{H})
  \end{equation}
\end{theorem}
\begin{proof}
  Let $\mathrm{q}(t): \mathbb{R} \to \mathrm{Spin}(3)$ with $\mathrm{q}(0)=1$. Since $\mathrm{q}\in\mathrm{Spin}(3)$ is where the quadrance is equal to $1$.
  \begin{equation}
    \mathrm{q}\mathrm{q}^*=1 \nonumber
  \end{equation}
  The derivative w.r.t. $t$ is then
  \begin{equation}
    \mathrm{\dot{q}}\mathrm{q}^* + \mathrm{q}\mathrm{\dot{q}}^* = 0. \nonumber
  \end{equation}
  At $t=0$
  \begin{equation}
    \mathrm{\dot{q}} + \mathrm{\dot{q}}^* = 0 \nonumber
  \end{equation}
  Expanding to the scalar plus vector form
  \begin{equation}
    \begin{pmatrix}
      \dot{\mathbf{q}} \\
      \dot{q}_0
    \end{pmatrix} +
    \begin{pmatrix}
      - \dot{\mathbf{q}} \\
      \dot{q}_0
    \end{pmatrix} = 0 \nonumber
  \end{equation}
  This is only true when $\dot{q}_0=0$, so $\dot{q}$ is a pure quaternion belonging to $\mathrm{Im}(\mathbb{H})$.
\end{proof}

\begin{theorem}
  The Lie algebra that corresponds to the Rotation group is a group of adjoint operators, such that
  \begin{equation}
    \mathfrak{rot}(3) := \left\{\ad_{\mathrm{\dot{q}}}: \mathbf{v} \to [\mathrm{\dot{q}}, (\mathbf{v}, 0)], \mathrm{\dot{q}}\in\mathfrak{spin}(3)\right\}.
  \end{equation}
\end{theorem}
\begin{proof}
  Let $R_\mathrm{q}(t): \mathbb{R} \to \mathrm{Rot}(3)$ with $R_\mathrm{q}(t=0)=R_\mathrm{1}$.
  Take the derivative of the action, fixing $\dot{\mathbf{v}}=0$
  \begin{equation}
    \frac{d}{dt} (\mathrm{q} \otimes (\mathbf{v}, 0) \otimes \mathrm{q}^*) = \mathrm{\dot{q}}\otimes (\mathbf{v}, 0) \otimes \mathrm{q}^* + \mathrm{q} \otimes (\mathbf{v}, 0) \otimes \mathrm{\dot{q}}^* \nonumber
  \end{equation}
  At $t=0$,
  \begin{equation}
    \frac{d}{dt} (\mathrm{q} \otimes (\mathbf{v}, 0) \otimes \mathrm{q}^*) = \mathrm{\dot{q}}\otimes (\mathbf{v}, 0) + (\mathbf{v}, 0) \otimes \mathrm{\dot{q}}^*\nonumber
  \end{equation}
  Since $\mathrm{q}\in\mathrm{Spin}(3)$, $\mathrm{\dot{q}}\in\mathfrak{spin}(3)=\mathrm{Im}(\mathbb{H})$, so $\dot{q}^*=-\dot{q}$, giving
  \begin{equation}
    \frac{d}{dt} (\mathrm{q} \otimes (\mathbf{v}, 0) \otimes \mathrm{q}^*) = \mathrm{\dot{q}}\otimes (\mathbf{v}, 0) - (\mathbf{v}, 0) \otimes \mathrm{\dot{q}}, \nonumber
  \end{equation}
  which is the commutator
  \begin{equation}
    \frac{d}{dt} (\mathrm{q} \otimes (\mathbf{v}, 0) \otimes \mathrm{q}^*) = [\mathrm{\dot{q}}, (\mathbf{v}, 0)] \nonumber
  \end{equation}
  We define the \emph{adjoint operator}
  \begin{equation}
    \ad_\mathrm{\dot{q}}(\mathrm{p}) = \mathrm{p} \to [\mathrm{\dot{q}}, \mathrm{p}] \nonumber
  \end{equation}
  which gives us our result
  \begin{equation}
    \frac{d}{dt} (\mathrm{q} \otimes (\mathbf{v}, 0) \otimes \mathrm{q}^*) = \ad_\mathrm{\dot{q}}(\mathbf{v}, 0) \nonumber
  \end{equation}
\end{proof}

\begin{theorem}
  The differential equation
  \begin{equation}
    \mathrm{\dot{q}} = \frac{1}{2}
    \boldsymbol\omega \otimes
    \mathrm{q}
  \end{equation}
  governs the evolution of instantaneous update of quaternion rotation.
\end{theorem}
\begin{proof}
  Let $\mathrm{q}: t \mapsto \mathrm{q}(t)\in\mathrm{Spin}(3)$.
  At any time $t$
  \begin{equation}
    \mathrm{\dot{q}}\mathrm{q}^* + \mathrm{q}\mathrm{\dot{q}}^* = 0. \nonumber
  \end{equation}
  Since, conjugation reverses the order of multiplication
  \begin{equation}
    \mathrm{\dot{q}}\mathrm{q}^* + (\mathrm{\dot{q}}\mathrm{q}^*)^* = 0. \nonumber
  \end{equation}
  Therefore, $\mathrm{\dot{q}}\mathrm{q}^*$ is a pure quaternion.
  Define the angular velocity to be $\boldsymbol\omega = 2\mathrm{\dot{q}}\mathrm{q}^*\in\mathrm{Im}(\mathbb{H})$ (see remark below).
  Right multiply both sides by $\mathrm{q}$
  \begin{equation}
    \mathrm{\dot{q}} = \boldsymbol\omega \otimes \mathrm{q}, \nonumber
  \end{equation}
  which is double what we want...
\end{proof}

\begin{remark}
  Using $\boldsymbol\Omega:=\mathrm{\dot{q}}\mathrm{q}^*$ leads to double the angle during integration.
\end{remark}
\begin{proof}
  Consider a simply rotating vector with
  \begin{equation}
    \mathbf{v}(t) := \mathrm{q}(t) \otimes \mathbf{v}_0 \otimes \mathrm{q}^*(t). \nonumber
  \end{equation}
  Take the derivative of the dot product
  \begin{equation}
    \frac{d}{dt}(\mathbf{v}(t) \cdot \mathbf{v}_0) = \dot{\mathbf{v}} \cdot \mathbf{v}_0
  \end{equation}
  Since $\ad_\Omega \in \mathfrak{rot}(3)$ is a generator for the map
  \begin{equation}
    \dot{\mathbf{v}} \cdot \mathbf{v}_0 = \left[\Omega, \mathbf{v}_0\right] \cdot \mathbf{v}_0 \nonumber
  \end{equation}
  Or using our commutator relation
  \begin{equation}
    \dot{\mathbf{v}} \cdot \mathbf{v}_0 = 2(\Omega \times \mathbf{v}_0) \cdot \mathbf{v}_0 \nonumber
  \end{equation}
  Using the triple product formula
  \begin{equation}
    \dot{\mathbf{v}} \cdot \mathbf{v}_0 = 2\Omega \cdot \left(\mathbf{v}_0 \times \mathbf{v}_0\right) \nonumber
  \end{equation}
  which is just
  \begin{equation}
    \dot{\mathbf{v}} \cdot \mathbf{v}_0 = \mathrm{0} \nonumber
  \end{equation}
\end{proof}
