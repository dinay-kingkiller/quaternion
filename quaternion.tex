\title{Quaternion Rotation}

\begin{abstract}
  Quaternions provide a unified framework that can be interpreted as elements of a division ring, hyper-complex numbers, or a four-dimensional vector space.
  Despite extensive literature, definitions and conventions for quaternion operations vary, which can complicate the translation of results into computational applications.
  This paper presents a formal treatment of quaternion algebra, defining addition, multiplication, and conjugation consistently, and demonstrates their use in representing rotations.
  We emphasize a formulation suitable for direct implementation in code, laying the foundation for extending these methods to rigid-body transformations in $\mathrm{SE}(3)$.
  Proofs of basic quaternion properties are summarized, while the focus remains on applications and computational implementation.
\end{abstract}

\section{The Quaternion Ring}
Quaternions arise as part of a remarkable sequence of algebraic extensions of the real numbers.
Starting from the reals $\mathbb{R}$ and complex numbers $\mathbb{C}$, one can ask how far the familiar properties of arithmetic can be preserved in higher dimensions.
The quaternions $\mathbb{H}$ are the smallest extension in which division remains possible but commutativity of multiplication must be abandoned.
Beyond this point, even more fundamental properties, such as associativity, begin to fail, as seen in the octonions.
\begin{definition}[Quaternion]
  Formally, a quaternion is an ordered 4-tuple.
  \begin{equation}
    \mathbf{q} = (x, y, z, w) \in \mathbb{H} \subset \mathbb{R}^4.
  \end{equation}
  The set of quaternions $\mathbb{H}$, equipped with the operations $\oplus$ and $\otimes$, forms a division ring.
\end{definition}

\begin{definition}[Quaternion Addition]
  For two quaternions, addition is defined componentwise
  \begin{equation}
    \begin{pmatrix}
      x_1 \\
      y_1 \\
      z_1 \\
      w_1
    \end{pmatrix} \oplus
    \begin{pmatrix}
      x_2 \\
      y_2 \\
      z_2 \\
      w_2
    \end{pmatrix} =
    \begin{pmatrix}
      x_1 + x_2 \\
      y_1 + y_2 \\
      z_1 + z_2 \\
      w_1 + w_2
    \end{pmatrix}
  \end{equation}
\end{definition}

\begin{definition}[Quaternion Multiplication]
  Multiplication is defined as
  \begin{equation} \label{eqn:multiplication}
    \begin{pmatrix}
      x_1 \\
      y_1 \\
      z_1 \\
      w_1
    \end{pmatrix} \otimes 
    \begin{pmatrix}
      x_2 \\
      y_2 \\
      z_2 \\
      w_2
    \end{pmatrix} =
    \begin{pmatrix}
      w_1x_2 + x_1w_2 + y_1z_2 - z_1y_2 \\
      w_1y_2 - x_1z_2 + y_1w_2 + z_1x_2 \\
      w_1z_2 + x_1y_2 - y_1x_2 + z_1w_2 \\
      w_1w_2 - x_1x_2 - y_1y_2 - z_1z_2
    \end{pmatrix}
  \end{equation}
  This definition aligns with the conventions used in the ROS package \texttt{tf2}.
  While other sources may define quaternion multiplication differently, they share the same underlying algebraic structure.
  An equivalent formulation appears in the next section.
\end{definition}

\begin{theorem}
  Quaternions form a commutative group under addition:
  \begin{itemize}
    \item Associativity: $\mathbf{q}_1 \oplus (\mathbf{q}_2 \oplus \mathbf{q}_3) = (\mathbf{q}_1 \oplus \mathbf{q}_2) \oplus \mathbf{q}_3$
    \item Commutativity: $\mathbf{q}_1 \oplus \mathbf{q}_2 = \mathbf{q}_2 \oplus \mathbf{q}_1$
    \item Identity: $\mathbf{q} \oplus \mathbf{0} = \mathbf{q} = \mathbf{0} \oplus \mathbf{q}$ where $\mathbf{0}$ is the 0-quaternion $(0, 0, 0, 0)$
    \item Inverse: $\mathbf{q} \oplus (-\mathbf{q}) = \mathbf{q} \ominus \mathbf{q} = \mathbf{0} =(-\mathbf{q}) \oplus \mathbf{q}$ where the inverse is defined $-\mathbf{q}=(-x, -y, -z, -w)$
  \end{itemize}
\end{theorem}

\begin{theorem}
  The (nonzero) quaternions form a \emph{non-commutative} group under multiplication, which gives it the following properties
  \begin{itemize}
    \item Associativity: $\mathbf{q}_1 \otimes (\mathbf{q}_2 \otimes \mathbf{q}_3) = (\mathbf{q}_1 \otimes \mathbf{q}_2) \otimes \mathbf{q}_3$
    \item Identity: $\mathbf{q} \otimes \mathbf{1} = \mathbf{q} = \mathbf{1} \otimes \mathbf{q}$ where $\mathbf{1}$ is the identity quaternion $(0, 0, 0, 1)$
  \item Inverse: $\mathbf{q} \otimes \mathbf{q}^{-1}=\mathbf{q} / \mathbf{q} = \mathbf{1} = \mathbf{q}^{-1} \otimes \mathbf{q}$ unless $q$ is the zero quaternion.
  \end{itemize}
\end{theorem}

\begin{theorem}
  With the combination of operations, quaternions form a division ring, which adds distributive properties
  \begin{itemize}
    \item Left distributivity: $\mathbf{q}_1 \otimes \left(\mathbf{q}_2 \oplus \mathbf{q}_3\right) = \left(\mathbf{q}_1 \otimes \mathbf{q}_2\right) \oplus \left(\mathbf{q}_1 \otimes \mathbf{q}_3\right)$
    \item Right distributivity: $\left(\mathbf{q}_1 \oplus \mathbf{q}_2\right) \otimes \mathbf{q}_3 = \left(\mathbf{q}_1 \otimes \mathbf{q}_3\right) \oplus \left(\mathbf{q}_2 \otimes \mathbf{q}_3\right)$
  \end{itemize}
\end{theorem}

\section{The Quaternion Space}
While the component form is convenient for computation, it obscures the algebraic structure.
By introducing a basis, quaternion multiplication can be expressed in a more mnemonic and structured form.
Scalar multiplication equips quaternions with the structure of a vector space over the real numbers.
\begin{definition}[Quaternion Scaling]
  Scaling a quaternion by a real number is defined componentwise.
  \begin{equation}
    s
    \begin{pmatrix}
      x \\
      y \\
      z \\
      w
    \end{pmatrix} =
    \begin{pmatrix}
      sx \\
      sy \\
      sz \\
      sw
    \end{pmatrix}
  \end{equation}
\end{definition}

\begin{theorem}
  Quaternions form a vector space over the real numbers, which means they have the following properties
  \begin{itemize}
  \item $(\mathbb{H}, \oplus)$ forms a commutative group.
    The details of which are defined in the previous section.
  \item Compatibility: $s_1(s_2\mathbf{q}) = (s_1s_2)\mathbf{q}$
  \item Scalar Identity: $1\mathbf{q} = \mathbf{q}$ where $1\in\mathbb{R}$
  \item Distributivity over quaternion addition: $s(\mathbf{q}_1+ \mathbf{q}_2) = s\mathbf{q}_1 + s\mathbf{q}_2$
  \item Distributivity over scalar multiplication: $(s_1+ s_2) \mathbf{q} = s_1\mathbf{q} + s_2\mathbf{q}$
  \end{itemize}
  In addition, quaternion and scalar multiplication are compatible
  \begin{equation}
    s \mathbf{q} = (0, 0, 0, s) \otimes \mathbf{q} \nonumber
  \end{equation}
\end{theorem}

\begin{definition}
  In the quaternion vector space, define four orthogonal unit basis elements
  \begin{equation}
    \mathbf{i} =
    \begin{pmatrix}
      1 \\
      0 \\
      0 \\
      0
    \end{pmatrix},
    \mathbf{j} =
    \begin{pmatrix}
      0 \\
      1 \\
      0 \\
      0
    \end{pmatrix}, 
    \mathbf{k} =
    \begin{pmatrix}
      0 \\
      0 \\
      1 \\
      0
    \end{pmatrix}, 
    \mathbf{1} =
    \begin{pmatrix}
      0 \\
      0 \\
      0 \\
      1
    \end{pmatrix}.
  \end{equation}
  Every quaternion can be described in this basis
  \begin{equation}
    q = x \mathbf{i} + y \mathbf{j} + z \mathbf{k} + w \mathbf{1}
  \end{equation}
  This representation is equivalent to the component form $(x, y, z, w)$.
\end{definition}

\begin{theorem}
  The definition of quaternion multiplication in equation \ref{eqn:multiplication} is not particularly mnemonic.
  Historically quaternions were used to extend complex numbers and as such, using the bases definition above, the complex $i^2=-1$ can be extended for quaternion hypercomplex numbers
  \begin{equation}
    \mathbf{i}^2 = \mathbf{j}^2 = \mathbf{k}^2 = \mathbf{ijk} = -\mathbf{1}
  \end{equation}
  Some papers use different definitions of unit bases, which would lead to a different equation \ref{eqn:multiplication}.
  The algebraic structure remains equivalent across differing conventions, but comparing sources can introduce sign errors or apparent changes in commutativity.
\end{theorem}
While this basis formulation is compact and mnemonic, an equivalent formulation in terms of vector operations is more useful for geometric interpretation and computation.

\section{Pure Quaternions}
In the previous section we said quaternions form a vector space, so technically every quaternion is a 4-vector.
In practice, quaternions are particularly useful because of how they interact with 3-vectors.
\begin{definition}[Vector Quaternion]
  Quaternions can be interpreted as vectors with an attached scalar component.
  Using a slight abuse of notation, every quaternion can be decomposed into a \emph{vector} (or \emph{pure quaternion}) $\vec{v} = \left(x, y, z, 0\right)$ and a scalar quaternion $s = \left(0,0,0,s\right)$: either $\vec{v}+s$ or, like below, $(\vec{v}, s)$.
  Addition is defined as
  \begin{equation}
    \begin{pmatrix}
      \vec{v}_1 \\
      s_1
    \end{pmatrix}
    \oplus
    \begin{pmatrix}
      \vec{v}_2 \\
      s_2
    \end{pmatrix}
    =
    \begin{pmatrix}
      \vec{v}_1 + \vec{v}_2\\
      s_1 + s_2
    \end{pmatrix} \nonumber
  \end{equation}
  and multiplication can be defined from the vector dot and cross product
  \begin{equation}
    \begin{pmatrix}
      \vec{v}_1 \\
      s_1
    \end{pmatrix}
    \otimes
    \begin{pmatrix}
      \vec{v}_2 \\
      s_2
    \end{pmatrix}
    =
    \begin{pmatrix}
      s_1\vec{v}_2 + s_2\vec{v}_1 + \vec{v}_1 \times \vec{v}_2 \\
      s_1s_2 - \vec{v}_1 \cdot \vec{v}_2
    \end{pmatrix} \nonumber
  \end{equation}
  For pure quaternions (a.k.a. vectors), that multiplication becomes
  \begin{equation}
    \begin{pmatrix}
      \vec{v}_1 \\
      0
    \end{pmatrix}
    \otimes
    \begin{pmatrix}
      \vec{v}_2 \\
      0
    \end{pmatrix} =
    \begin{pmatrix}
      \vec{v}_1 \times \vec{v}_2\\
      -\vec{v}_1 \cdot \vec{v}_2
    \end{pmatrix} \nonumber
  \end{equation}
  This identifies quaternion multiplication as a combination of the familiar dot and cross products on $\mathbb{R}^3$.
  The scalar part is minus the dot product of the two vectors and the vector part is the cross product, another way to re-derive equation \ref{eqn:multiplication} when on a subway with no internet.
\end{definition}

\begin{theorem}[Quaternion Commutator]
  Quaternion multiplication is not commutative.
  Its commutator is given by
  \begin{equation}
    \begin{pmatrix}
      \vec{v}_1 \\
      s_1
    \end{pmatrix} \otimes
    \begin{pmatrix}
      \vec{v}_2 \\
      s_2
    \end{pmatrix} -
    \begin{pmatrix}
      \vec{v}_2 \\
      s_2
    \end{pmatrix} \otimes
    \begin{pmatrix}
      \vec{v}_1 \\
      s_1
    \end{pmatrix} =
    \begin{pmatrix}
      2\vec{v}_1 \times \vec{v}_2 \\
      0
    \end{pmatrix} \label{eqn:commutator}
  \end{equation}
\end{theorem}
\begin{proof}
  For $\mathbf{q}_1=(\vec{v}_1, s_1)$ and $\mathbf{q}_2=(\vec{v}_2, s_2)$, the commutator expands to
  \begin{equation}
    \begin{pmatrix}
      \vec{v}_1 \\
      s_1
    \end{pmatrix} \otimes
    \begin{pmatrix}
      \vec{v}_2 \\
      s_2
    \end{pmatrix} -
    \begin{pmatrix}
      \vec{v}_2 \\
      s_2
    \end{pmatrix} \otimes
    \begin{pmatrix}
      \vec{v}_1 \\
      s_1
    \end{pmatrix} =
    \begin{pmatrix}
      s_1\vec{v}_2 + s_2\vec{v}_1 + \vec{v}_1 \times \vec{v}_2 \\
      s_1s_2 - \vec{v}_1 \cdot \vec{v}_2
    \end{pmatrix} -
    \begin{pmatrix}
      s_2\vec{v}_1 + s_1\vec{v}_2 + \vec{v}_2 \times \vec{v}_1 \\
      s_2s_1 - \vec{v}_2 \cdot \vec{v}_1
    \end{pmatrix} \nonumber
  \end{equation}
  Since only the cross product is \emph{not} commutative, the commutator reduces down to
  \begin{equation}
    \begin{pmatrix}
      \vec{v}_1 \\
      s_1
    \end{pmatrix} \otimes
    \begin{pmatrix}
      \vec{v}_2 \\
      s_2
    \end{pmatrix} -
    \begin{pmatrix}
      \vec{v}_2 \\
      s_2
    \end{pmatrix} \otimes
    \begin{pmatrix}
      \vec{v}_1 \\
      s_1
    \end{pmatrix} =
    \begin{pmatrix}
      \vec{v}_1\times\vec{v}_2-\vec{v}_2\times\vec{v}_1 \\
      0
    \end{pmatrix} \nonumber
  \end{equation}
  But, because the cross product is \emph{anti-}commutative
  \begin{equation}
    \begin{pmatrix}
      \vec{v}_1 \\
      s_1
    \end{pmatrix} \otimes
    \begin{pmatrix}
      \vec{v}_2 \\
      s_2
    \end{pmatrix} -
    \begin{pmatrix}
      \vec{v}_2 \\
      s_2
    \end{pmatrix} \otimes
    \begin{pmatrix}
      \vec{v}_1 \\
      s_1
    \end{pmatrix} =
    \begin{pmatrix}
      2\vec{v}_1\times\vec{v}_2\\
      0
    \end{pmatrix} \nonumber
  \end{equation}
\end{proof}

\section{Unit Quaternion Subgroup}
The algebra of quaternions admits a natural notion of conjugation, which in turn defines a quadratic quantity known as the quadrance.
Together, these structures endow $\mathbb{H}$ with the structure of a \emph{composition algebra}, in which the quadrance is multiplicative under quaternion multiplication.
This multiplicative property allows us to identify a distinguished subset of quaternions whose quadrance is unity.
As we will see, this subset forms a group under quaternion multiplication and will play a central role in describing rotations in three-dimensional space.

\begin{definition}
  For a quaternion $\mathbf{q}=(\vec{v}, s)$, its \emph{conjugate} is given by 
  \begin{equation}
    \mathbf{q}^* := (-\vec{v}, s)
  \end{equation}
\end{definition}

\begin{definition}
  The \emph{quadrance} of a quaternion is the map $Q: \mathbb{H} \to \mathbb{R}$ defined by
  \begin{equation}
    Q(\mathbf{q}) := \mathbf{q} \otimes \mathbf{q}^*.
  \end{equation}
  Or, in coordinate form
  \begin{equation}
    Q(x\mathbf{i}+y\mathbf{j}+z\mathbf{k}+w) = x^2 + y^2 + z^2 + w^2.
  \end{equation}
\end{definition}

\begin{theorem}
  With these new definitions, the quaternion algebra $\mathbb{H}$ forms a composition algebra, which has the following properties:
  \begin{itemize}
    \item Conjugation Involution: $\left(\mathbf{q}^*\right)^* = \mathbf{q}$,
    \item Conjugation Reverses Multiplication: $(\mathbf{p}\otimes\mathbf{q})^*=\mathbf{q}^*\otimes\mathbf{p}^*$,
    \item Quadrance Nondegeneracy: $Q(\mathbf{q})=0 \Leftrightarrow \mathbf{q} = \mathbf{0}$,
    \item Quadrance Compatibility with Multiplication: $Q(\mathbf{p})Q(\mathbf{q})=Q(\mathbf{p}\otimes \mathbf{q})$.
  \end{itemize}
\end{theorem}

\begin{definition}
  The \emph{unit quaternion}, or \emph{versor}, subgroup $\mathrm{Spin}(3) \subset (\mathbb{H}, \otimes)$, is the set of quaternions with quadrance equal to 1.
  \begin{equation}
    \mathrm{Spin}(3) := \left\{\mathbf{q} \in \mathbb{H} : Q(\mathbf{q}) = 1\right\}
  \end{equation}
\end{definition}

\begin{theorem}
  This group has the following group properties
  \begin{itemize}
    \item Closure: If $Q(\mathbf{p})=1$ and $Q(\mathbf{q})=1$, then $Q(\mathbf{p}\otimes\mathbf{q})=1$.
    \item Identity: The identity quaternion is a versor $\mathbf{1}\in\mathrm{Spin}(3)$.
    \item Inverse: The inverse of a versor is the conjugate which is also a versor $\mathbf{q} \otimes \mathbf{q}^* = \mathbf{1}$.
  \end{itemize}
\end{theorem}

\section{Quaternion Rotation Group}

\begin{definition}
  Unit quaternions act on vectors by conjugation, producing rotations in three-dimensional space.
  The set of transforms of the form below form the rotation group.
  \begin{equation}
    \mathrm{Rot}(3) := \{\mathrm{R}_{\mathbf{q}} :\mathbf{q}\in\mathrm{Spin}(3), \vec{v} \mapsto \mathbf{q}\otimes(\vec{v}, 0)\otimes\mathbf{q}^*\}
  \end{equation}
\end{definition}

\begin{theorem}
  These transformations form a group
  \begin{itemize}
    \item Closure: Composition is quaternion multiplication $R_\mathbf{p} \circ R_\mathbf{q} = R_{\mathbf{p}\otimes\mathbf{q}}$
  \item Identity: The identity quaternion forms the identity rotation $\vec{v} = R_\mathbf{1}(\vec{v})$
  \item Inverse: The conjugate forms the inverse rotation $R_\mathbf{q} \circ R_{\mathbf{q}^*} = R_\mathbf{1}$
  \item Cover: For all vectors where $Q(\vec{u})=Q(\vec{v})$ there is a rotation $\vec{u}=R_\mathbf{q}(\vec{v})$
  \item Double Cover: Rotations are not unique $R_\mathbf{q}(\vec{v})=R_{-\mathbf{q}}(\vec{v})$
  \end{itemize}
\end{theorem}

\begin{theorem}
  In addition, quaternion rotation behaves well with 3-vectors:
  \begin{itemize}
    \item Orthogonality: $Q(R_\mathbf{q}(\vec{v}))=Q(v)$
    \item Preserves the dot product: $\vec{u} \cdot \vec{v}=R_\mathbf{q}(\vec{u}) \cdot R_\mathbf{q}(\vec{v})$
    \item Preserves the cross product: $R_\mathbf{q}\left(\vec{u}\times\vec{v}\right)=R_\mathbf{q}\vec{u} \times R_\mathbf{q}\vec{v}$.
  \end{itemize}
\end{theorem}

\begin{theorem}[Quaternion to Matrix]
  Every rotation $\mathrm{A}\to\mathrm{B}$ can be described as both a quaternion and a rotation matrix
  \begin{equation}
    R\vec{v} = R_\mathbf{q} \left(\vec{v}\right)
  \end{equation}
  that rotation matrix also collects the dot products of the rotation.
  
  \begin{tabular}{c|ccc}
    $R_{A2B}$ & $\hat{\mathbf{a}}_x$ & $\hat{\mathbf{a}}_y$ & $\hat{\mathbf{a}}_z$ \\
    \hline
    $\hat{\mathbf{b}}_x$ & $1-2(y^2+z^2)$ & $2(xy-zw)$ & $2(xz+yw)$ \\
    $\hat{\mathbf{b}}_y$ & $2(xy+zw)$ & $1-2(x^2+z^2)$ & $2(yz-xw)$ \\
    $\hat{\mathbf{b}}_z$ & $2(xz-wy)$ & $2(xw+yz)$ & $1-2(x^2+y^2)$
  \end{tabular}\centering
\end{theorem}
\begin{proof}
  $R$ can be formed from algebra.
  \begin{equation}
    \begin{pmatrix}
      R & \vec{0}^\top \\
      \vec{0} & 0
    \end{pmatrix}
    \begin{pmatrix}
      v_x \\
      v_y \\
      v_z \\
      0
    \end{pmatrix}
    =
    \begin{pmatrix}
      x \\
      y \\
      z \\
      w
    \end{pmatrix} \otimes
    \begin{pmatrix}
      v_x \\
      v_y \\
      v_z \\
      0
    \end{pmatrix}
    \otimes
    \begin{pmatrix}
      -x \\
      -y \\
      -z \\
      w
    \end{pmatrix} \nonumber
  \end{equation}
  Expanding and collecting like terms
  \begin{equation}
    R =
    \begin{pmatrix}
      x^2-y^2-z^2+w^2 & 2xy-2zw & 2xz+2yw \\
      2xy+2zw & -x^2+y^2-z^2+w^2 & -2xw+2yz \\
      2xz-2wy & 2xw+2yz & -x^2-y^2+z^2+w^2
    \end{pmatrix} \nonumber
  \end{equation}
  Since this is a unit quaternion, its quadrance is equal to one: $1=x^2+y^2+z^2+w^2$.
  \begin{equation}
    R =
    \begin{pmatrix}
      1-2(y^2+z^2) & 2(xy-zw) & 2(xz+yw) \\
      2(xy+zw) & 1-2(x^2+z^2) & 2(yz-xw) \\
      2(xz-wy) & 2(xw+yz) & 1-2(x^2+y^2)
    \end{pmatrix} \nonumber
  \end{equation}
\end{proof}

\section{Quaternion Rotation Algebra}
\begin{theorem}
  If frame $B$ is rotating in frame $A$ at a constant angular velocity, then the rate of change of the quaternion is given by
  \begin{equation}
    \mathrm{\dot{q}} = \frac{1}{2}
    \begin{pmatrix}
      \boldsymbol\omega\\
      0
    \end{pmatrix}
    \otimes \mathrm{q}
  \end{equation}
  where $\boldsymbol\omega\in\mathbb{R}^3$ is the angular velocity of frame $B$ in $A$.
\end{theorem}
\begin{proof}
  A unit quaternion $\mathrm{q}$ describes a rotation of a vector via
  \begin{equation}
    \begin{pmatrix}
      \mathbf{v}'\\
      0
    \end{pmatrix}
    = \mathrm{q} \otimes
    \begin{pmatrix}
      \mathbf{v} \\
      0
    \end{pmatrix}
    \otimes \mathrm{q}^* \nonumber
  \end{equation}
  where $\mathbf{v}$ and $\mathbf{v}'$ is that vector described in a basis $A$ and $B$ respectively.
  Let $\boldsymbol\omega$ be the angular velocity of reference frame $B$ in $A$. Then by the Transport Theorem, the derivative in basis $A$ is
  \begin{equation}
    \frac{d}{dt} \mathbf{v}' = \frac{d}{dt} \mathbf{v} + \boldsymbol\omega \times \mathbf{v}' \nonumber
  \end{equation}
  affix $\mathbf{v}$ in $A$ (i.e. $\frac{d}{dt}\mathbf{v}=0$).
  Using that and expanding the derivative of $\mathbf{v}'$ gives
  \begin{equation}
    \mathrm{\dot{q}} \otimes
    \begin{pmatrix}
      \mathbf{v} \\
      0
    \end{pmatrix}
    \otimes \mathrm{q}^* + \mathrm{q} \otimes
    \begin{pmatrix}
      \mathbf{v} \\
      0
    \end{pmatrix}
    \otimes \mathrm{\dot{q}^*} =
    \begin{pmatrix}
      \boldsymbol\omega\times\mathbf{v}' \\
      0
    \end{pmatrix} \nonumber
  \end{equation}
  Or in terms of $\mathbf{v}'$
  \begin{equation}
    \mathrm{\dot{q}}\otimes\mathrm{q^*}\otimes
    \begin{pmatrix}
      \mathbf{v}'\\
      0
    \end{pmatrix}
    +
    \begin{pmatrix}
      \mathbf{v}'\\
      0
    \end{pmatrix}
    \otimes \mathrm{q} \otimes \mathrm{\dot{q}^*} =
    \begin{pmatrix}
      \boldsymbol\omega\times\mathbf{v}' \\
      0
    \end{pmatrix} \nonumber
  \end{equation}
  Since the conjugate is the inverse of the quaternion
  \begin{equation}
    \mathrm{q}\otimes\mathrm{q^*} = 1 \implies
    \mathrm{\dot{q}} \otimes\mathrm{q^*}+\mathrm{q}\otimes\mathrm{\dot{q}^*} = 0 \implies
    \left(\mathrm{\dot{q}}\otimes\mathrm{q}^*\right)^*=-\mathrm{\dot{q}}\otimes \mathrm{q^*} \nonumber
  \end{equation}
  which implies that $\mathrm{\dot{q}}\otimes\mathrm{q}^*$ is a pure quaternion (For a quaternion $\mathrm{p}$, $-\mathrm{p}=\mathrm{p^*}$ is only true for a quaternion with a zero scalar part). Let $(\mathbf{u}, 0) = \mathrm{\dot{q}}\otimes\mathrm{q}^*=-\mathrm{q}\otimes\mathrm{\dot{q}^*}$, then
  \begin{equation}
    \begin{pmatrix}
      \mathbf{u} \\
      0
    \end{pmatrix}
    \otimes
    \begin{pmatrix}
      \mathbf{v}'\\
      0
    \end{pmatrix}
    -
    \begin{pmatrix}
      \mathbf{v}'\\
      0
    \end{pmatrix}
    \otimes
    \begin{pmatrix}
      \mathbf{u}\\
      0
    \end{pmatrix}
    =
    \begin{pmatrix}
      \boldsymbol\omega\times\mathbf{v}'\\
      0
    \end{pmatrix} \nonumber
  \end{equation}
  The left-hand side is the quaternion commutator. Expanded it becomes
  \begin{equation}
    \begin{pmatrix}
      \mathbf{u} \\
      0
    \end{pmatrix}
    \otimes
    \begin{pmatrix}
      \mathbf{v}'\\
      0
    \end{pmatrix}
    -
    \begin{pmatrix}
      \mathbf{v}'\\
      0
    \end{pmatrix}
    \otimes
    \begin{pmatrix}
      \mathbf{u}\\
      0
    \end{pmatrix}
    =
    \begin{pmatrix}
      \mathbf{u}\times\mathbf{v}'-\mathbf{v}'\times\mathbf{u} \\
      \mathbf{u}\cdot\mathbf{v}'-\mathbf{v}'\cdot\mathbf{u}
    \end{pmatrix}
    =
    \begin{pmatrix}
      2\mathbf{u}\times\mathbf{v}'\\
      0
    \end{pmatrix} \nonumber
  \end{equation}
  The vector part of the equation then becomes
  \begin{equation}
    2\mathbf{u}\times\mathbf{v}' = \boldsymbol\omega\times\mathbf{v}' \nonumber
  \end{equation}
  which only implies that $(2\mathbf{u}-\boldsymbol\omega)\times\mathbf{v}' = \mathbf{0}$.
  The relationship $2\mathbf{u}=\boldsymbol\omega$ is true only if $2\mathbf{u}\cdot\mathbf{v}'=\boldsymbol\omega\cdot\mathbf{v}'$.
  If the relationship is true then
  \begin{equation}
    2\mathrm{\dot{q}}\otimes\mathrm{q^*} =
    \begin{pmatrix}
      \boldsymbol\omega \\
      0
    \end{pmatrix} \nonumber
  \end{equation}
  Or re-arranging
  \begin{equation}
    \mathrm{\dot{q}} = \frac{1}{2}
    \begin{pmatrix}
      \boldsymbol\omega \\
      0
    \end{pmatrix}
    \otimes \mathrm{q} \nonumber
  \end{equation}
\end{proof}
